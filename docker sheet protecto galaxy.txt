# DOCKER SHEET PROYECTO DOCKER GALAXY

#DESCARGAR IMAGEN DE MYSQL
docker pull mysql:9.0
#CREAR CONTENEDOR MYSQL
docker run --name mysql-galaxy -p 3306:3306 -e MYSQL_ROOT_PASSWORD=123 -d mysql:9.0
#INGRESAR AL CONTENEDOR DE MYSQL
docker exec -it mysql-galaxy bash

#DESCARGAR IAMGEN DE MONGO
docker pull mongo:8.0

#CREAR CONTENEDOR DE MONGO
docker run --name mongo-galaxy -p 27017:27017 -d mongo:8.0
#INGRESAR AL CONTENEDOR DE MONGO
docker exec -it mongo-galaxy mongosh
